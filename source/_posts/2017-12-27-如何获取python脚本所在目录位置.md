```python
import os
# 获取文件路径
a = os.path.realpath(__file__)
# 将路径split开
b = os.path.split(a)
# 取第一个位置，即路径，第二个位置为文件名
c = b[0]

# 一行即
os.path.split(os.path.realpath(__file__))[0]
```
